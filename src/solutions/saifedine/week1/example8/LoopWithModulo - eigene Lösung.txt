public class LoopWithModulo {
     public static void main(String[] args) {
        for (int i = 0; i <=10 ; i += 2) {
            System.out.println(i);
        }

        int j = 0;
        while (j <= 10) {
            if (j % 2 == 0)
                System.out.println(j);
            j++;
        }
        }
    }
