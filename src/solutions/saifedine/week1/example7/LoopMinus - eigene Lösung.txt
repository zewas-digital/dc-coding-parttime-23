public class LoopMinus {
    public static void main(String[] args) {
        for (int i = 10; i > 0 ; i--) {
            System.out.println(i);
        }
        int i = 0;
            while (i <= 0){
                i--;
                System.out.println(i);

                if (i == -10){
                    break;

            }
        }
    }
}
